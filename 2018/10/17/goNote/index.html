2:多态


func Add(a, b int) int {


return a+b

}


func Minus(a, b int) int {


return a-b

}


// 定义函数类型
type funcType func(int ,int) int


// Calc参数为函数 返回值result int
func Calc(a, b int, fTest funcType) (result int) {


fmt.Println(“Calc”)

result = fTest(a, b)
return

}
func main() {


// 调用Calc函数 Add为函数
res := Calc(10, 20, Add)

fmt.Println(“res”, res)

result := Calc(20, 10, Minus)

fmt.Println(“result”, result)

}

func main() {


// 延时调用 main函数结束前调用
